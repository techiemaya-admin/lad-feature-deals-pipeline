2. Use React Hooks:

function MyComponent() {
  const { data, loading, error } = usePipelineBoard();
  
  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error.message}</div>;
  
  return <div>{data.stages.length} stages</div>;
}
